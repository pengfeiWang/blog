#插值法排序

```js
var arr = [10,20,33,8,5,22,19,90,2];
var len = arr.length;
var arr2 = [];
var flag = 0;
arr2[0] = arr[0]
for( var i = 0; i < len; i++ ) {
  var len2 = arr2.length;
  flag = 0;
  for( var j = 0; j < len2; j++ ) {
    if( arr2[j] > arr[i] ) {
      arr2.splice(j,0,arr[i]);
      flag = 1
      break;
    }
  }
  if( flag == 0 ) {
    arr2.push(arr[i])
  }
}
```
